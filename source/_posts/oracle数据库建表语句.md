```sql
CREATE TABLE TABLE_NAME (
    COLUMN DATATYPE;
    CONSTRAINT PK_TABLE_NAME PRIMARY KEY (PRIMARY_KEY)
);

COMMENT ON TABLE TABLE_NAME IS '表描述';
COMMENT ON COLUMN TABLE_NAME.COLUMN IS '字段描述';

CREATE SEQUENCE TABLE_NAME_SEQUENCE
INCREMENT BY 1
START WITH 100000000
MINVALUE 100000000
NOCYCLE
CACHE 100;

COMMIT;
```
